# user.collection

```javascript
user.collection([page])
```

Get this user's collections. By default, `page` is `1`.

For example:

```javascript
api.user('renfish')
    .collections()
    .then(console.log)
    .catch(console.trace)
```

the reuslt is:

It's also easy to get this user's all collections:

```javascript
var user = api.user('renfish')
var page = 1

next()

function next() {
    user.collections(page)
        .then(data => {
            if (data.length) {
                page++
                handleCollections(data)
                next()
            } else {
                console.log('Done!')
            }
        })
        .catch(console.trace)
}

function handleCollections(data) {
    // do something
}
```
