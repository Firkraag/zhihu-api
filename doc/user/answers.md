# user/org.answers

```javascript
user/org.answers([page])
```

Get answers that this user answered. By default, `page` is `1`.

For example:

```javascript
api.user('zhihuadmin')
    .answers()
    .then(console.log)
    .catch(console.trace)
```

It's also easy to get all answers that this user answered:

```javascript
var user = api.user('zhihuadmin')
var page = 1

next()

function next() {
    user.answers(page)
        .then(data => {
            if (data.length) {
                page++
                handleAnswers(data)
                next()
            } else {
                console.log('Done!')
            }
        })
        .catch(console.trace)
}

function handleAnswers(data) {
    // do something
}
```
